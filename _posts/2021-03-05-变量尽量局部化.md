### 变量局部化

```js
var i, str = "";
function nonLocal(){
    for(i = 0;i < str.length; i++){
    	console.log(1111);
	}
}
nonLocal();

//优化后
function local(){
    for(var i = 0;i < str.length; i++){
    	console.log(1111);
	}
}
```

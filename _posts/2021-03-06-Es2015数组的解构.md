### Es2015的解构

#### 数组的结构

```javascript
const arr = [100, 200, 300];
const [foo, bar, baz] = arr; //

const [, , baz] = arr; //

const [foo, ...rest] = arr; // ...res [200, 300]

const [foo, bar, baz = 123, more = 'default'] = arr;
console.log(more) //default
```



#### 对象的解构

```javascript
const { name } = obj
console.log(name)

//防止命名重复
const name = 'tom'
const { name: objName } = obj
console.log(objName)

//添加默认值
const name = 'tom'
const { name: objName = 'jack' } = obj
console.log(objName)

const { log } = console
log('foo')
log('bar')
log('123')
```
